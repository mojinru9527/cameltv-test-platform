## 3. 测试分析脚本准备

### 3.1 opencv.py（视频帧时间戳识别脚本）

在 `C:\直播测试脚本\` 下创建 `opencv.py` 文件，复制以下全部内容：

```python
import cv2
import pytesseract
import os
import sys
from datetime import datetime

def process_video(video_path):
    """
    处理视频文件，识别时间戳并生成日志
    """
    print(f"开始处理视频: {video_path}")

    if not os.path.exists(video_path):
        print(f"错误：文件 {video_path} 不存在")
        return

    cap = cv2.VideoCapture(video_path)

    if not cap.isOpened():
        print("错误：无法打开视频文件")
        return

    fps = cap.get(cv2.CAP_PROP_FPS)
    total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
    print(f"视频信息: FPS={fps}, 总帧数={total_frames}")

    log_file = video_path.replace('.mp4', '.log').replace('.mov', '.log')
    frame_count = 0
    success_count = 0

    with open(log_file, 'w', encoding='utf-8') as f:
        f.write("frame_number,timestamp,confidence\n")

        while True:
            ret, frame = cap.read()
            if not ret:
                break

            if frame_count % 10 == 0:
                gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

                # ⚠ 重要：根据实际视频时间戳位置调整ROI坐标
                height, width = gray.shape
                # [可调整参数] 时间戳区域坐标: y_start, y_end, x_start, x_end
                roi = gray[50:100, width-200:width-50]

                try:
                    timestamp = pytesseract.image_to_string(
                        roi, 
                        config='--psm 7 -c tessedit_char_whitelist=0123456789:.'
                    )
                    timestamp = timestamp.strip()
                    if timestamp and len(timestamp) > 0:
                        f.write(f"{frame_count},{timestamp},1.0\n")
                        success_count += 1
                        print(f"帧 {frame_count}: 识别到时间戳 {timestamp}")
                except Exception as e:
                    print(f"帧 {frame_count}: OCR识别失败 - {e}")

            frame_count += 1

            if frame_count % 100 == 0:
                progress = (frame_count / total_frames) * 100
                print(f"处理进度: {progress:.1f}%")

    cap.release()
    print(f"处理完成！成功识别 {success_count} 个时间戳")
    print(f"日志文件: {log_file}")

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("用法: python opencv.py <视频文件路径>")
        sys.exit(1)

    video_file = sys.argv[1]
    process_video(video_file)
```

### 3.2 analysis.py（延迟数据分析脚本）

在 `C:\直播测试脚本\` 下创建 `analysis.py` 文件，复制以下全部内容：

```python
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np
import sys
import os

def analyze_delay(log_file):
    """
    分析延迟日志文件
    """
    print(f"开始分析日志文件: {log_file}")

    try:
        df = pd.read_csv(log_file, encoding='utf-8')
    except UnicodeDecodeError:
        try:
            df = pd.read_csv(log_file, encoding='gbk')
        except:
            print("错误：无法读取日志文件，请检查编码格式")
            return None

    if df.empty:
        print("错误：日志文件为空")
        return None

    def parse_timestamp(ts):
        try:
            if ':' in str(ts):
                parts = str(ts).split(':')
                if len(parts) == 3:
                    return int(parts[0])*3600000 + int(parts[1])*60000 + float(parts[2])*1000
                elif len(parts) == 2:
                    return int(parts[0])*60000 + float(parts[1])*1000
            else:
                return float(ts) * 1000
        except:
            return None

    df['timestamp_ms'] = df['timestamp'].apply(parse_timestamp)
    df = df.dropna(subset=['timestamp_ms'])

    if df.empty:
        print("错误：没有有效的时间戳数据")
        return None

    base_time = df['timestamp_ms'].iloc[0]
    df['delay_ms'] = df['timestamp_ms'] - base_time

    # [可调整参数] 根据实际测试视频的帧率修改此值
    assumed_fps = 30
    df['expected_time'] = df['frame_number'] * (1000 / assumed_fps)
    df['actual_delay'] = df['delay_ms'] - df['expected_time']

    delays = df['actual_delay'].dropna()

    stats = {
        '数据点数': len(delays),
        '平均值(ms)': np.mean(delays),
        '中位数(ms)': np.median(delays),
        '最小值(ms)': np.min(delays),
        '最大值(ms)': np.max(delays),
        '标准差(ms)': np.std(delays),
        '95分位数(ms)': np.percentile(delays, 95)
    }

    print("\n=== 延迟统计结果 ===")
    for key, value in stats.items():
        if isinstance(value, float):
            print(f"{key}: {value:.2f}")
        else:
            print(f"{key}: {value}")

    # 生成图表
    plt.figure(figsize=(12, 8))

    plt.subplot(2, 1, 1)
    plt.plot(df['frame_number'], df['actual_delay'], 'b-', alpha=0.7, linewidth=1)
    plt.xlabel('帧号')
    plt.ylabel('延迟 (ms)')
    plt.title(f'视频延迟分析 - {os.path.basename(log_file)}')
    plt.grid(True, alpha=0.3)

    plt.subplot(2, 1, 2)
    plt.hist(delays, bins=50, alpha=0.7, edgecolor='black')
    plt.xlabel('延迟 (ms)')
    plt.ylabel('频次')
    plt.title('延迟分布直方图')
    plt.grid(True, alpha=0.3)

    plt.tight_layout()

    chart_file = log_file.replace('.log', '_chart.png')
    plt.savefig(chart_file, dpi=300, bbox_inches='tight')
    print(f"\n图表已保存: {chart_file}")

    result_file = log_file.replace('.log', '_results.txt')
    with open(result_file, 'w', encoding='utf-8') as f:
        f.write("视频延迟测试结果\n")
        f.write("================\n\n")
        for key, value in stats.items():
            if isinstance(value, float):
                f.write(f"{key}: {value:.2f}\n")
            else:
                f.write(f"{key}: {value}\n")

    print(f"详细结果已保存: {result_file}")
    return stats

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("用法: python analysis.py <日志文件路径>")
        sys.exit(1)

    log_file = sys.argv[1]
    if not os.path.exists(log_file):
        print(f"错误：文件 {log_file} 不存在")
        sys.exit(1)
    analyze_delay(log_file)
```

### 4.1 方案A：使用FFmpeg生成（快速，推荐新手）

```batch
@echo off

echo 1. 生成时间戳延迟测试视频...

  -vf "drawtext=text='TIMESTAMP: %%{pts}': fontcolor=white: fontsize=60: box=1: boxcolor=black@0.5: x=20: y=100, ^
       drawtext=text='TIME: %%{localtime}': fontcolor=white: fontsize=40: box=1: boxcolor=black@0.5: x=20: y=180, ^
       drawtext=text='FRAME: %%{n}': fontcolor=white: fontsize=40: box=1: boxcolor=black@0.5: x=20: y=240" ^
```

### 4.2 方案B：使用Python生成（高精度，适合进阶）

在 `C:\直播测试脚本\` 下创建 `generate_test_video.py`：

```python
import cv2
import numpy as np
import os

def generate_timestamp_video():
    """生成高精度时间戳测试视频"""
    output_dir = "C:\\直播测试素材"
    os.makedirs(output_dir, exist_ok=True)

    width, height = 1280, 720
    fps = 30
    duration = 300  # 5分钟
    total_frames = fps * duration

    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    video_path = os.path.join(output_dir, 'high_precision_timestamp.mp4')
    out = cv2.VideoWriter(video_path, fourcc, fps, (width, height))

    print("生成高精度时间戳测试视频...")

    for frame_num in range(total_frames):
        frame = np.zeros((height, width, 3), dtype=np.uint8)
        current_time_ms = (frame_num / fps) * 1000

        texts = [
            f"TIMESTAMP: {current_time_ms:.1f}ms",
            f"FRAME: {frame_num}",
            f"TIME: {frame_num//fps//60:02d}:{(frame_num//fps)%60:02d}.{(frame_num%fps)*33:03d}",
            f"TEST PATTERN - VIDEO DELAY MEASUREMENT"
        ]
        positions = [(50, 100), (50, 160), (50, 220), (50, 50)]
        colors = [(255, 255, 255), (255, 200, 100), (200, 255, 200), (100, 200, 255)]
        sizes = [1.5, 1.0, 0.8, 1.2]

        for i, text in enumerate(texts):
            cv2.putText(frame, text, positions[i], cv2.FONT_HERSHEY_SIMPLEX,
                       sizes[i], colors[i], 2)

        moving_x = (frame_num * 2) % (width - 200) + 100
        cv2.circle(frame, (moving_x, 500), 30, (0, 255, 0), -1)
        cv2.putText(frame, "MOVING MARKER", (moving_x-80, 550),
                   cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)

        if frame_num % 3 == 0:
            clock_time = f"CLOCK: {(current_time_ms//100)%10:01d}"
            cv2.putText(frame, clock_time, (width-200, 100),
                       cv2.FONT_HERSHEY_SIMPLEX, 1, (255, 100, 100), 2)

        out.write(frame)

        if frame_num % 300 == 0:
            progress = (frame_num / total_frames) * 100
            print(f"生成进度: {progress:.1f}%")

    out.release()
    print(f"视频生成完成: {video_path}")
    return video_path

def generate_fluency_test_video():
    """生成帧率测试视频"""
    output_dir = "C:\\直播测试素材"
    width, height = 1280, 720
    fps = 60
    duration = 600  # 10分钟

    fourcc = cv2.VideoWriter_fourcc(*'mp4v')
    video_path = os.path.join(output_dir, 'fluency_test_pattern.mp4')
    out = cv2.VideoWriter(video_path, fourcc, fps, (width, height))

    print("生成帧率测试视频...")

    for frame_num in range(fps * duration):
        frame = np.ones((height, width, 3), dtype=np.uint8) * 128

        bar_pos = (frame_num * 4) % (width + 200) - 100
        cv2.rectangle(frame, (bar_pos, 300), (bar_pos + 100, 400), (255, 255, 255), -1)

        info_texts = [
            "FLUENCY TEST PATTERN - 60FPS",
            f"Frame: {frame_num}",
            f"Time: {frame_num//fps//60:02d}:{(frame_num//fps)%60:02d}",
            "White bar should move smoothly"
        ]
        for i, text in enumerate(info_texts):
            cv2.putText(frame, text, (50, 100 + i*40),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 0, 0), 2)

        for x in range(0, width, 50):
            cv2.line(frame, (x, 0), (x, height), (200, 200, 200), 1)
        for y in range(0, height, 50):
            cv2.line(frame, (0, y), (width, y), (200, 200, 200), 1)

        out.write(frame)

    out.release()
    print(f"帧率测试视频生成完成: {video_path}")
    return video_path

if __name__ == "__main__":
    print("开始生成测试视频...")
    generate_timestamp_video()
    generate_fluency_test_video()
    print("所有测试视频生成完成！")
```
