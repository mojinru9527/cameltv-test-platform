python

import cv2

import pytesseract

import os

import sys

from datetime import datetime

def process\_video(video\_path):

"""

处理视频文件，识别时间戳并生成日志

"""

print(f"开始处理视频: {video\_path}")

# 检查文件是否存在

if not os.path.exists(video\_path):

print(f"错误：文件 {video\_path} 不存在")

return

# 打开视频文件

cap = cv2.VideoCapture(video\_path)

if not cap.isOpened():

print("错误：无法打开视频文件")

return

# 获取视频信息

fps = cap.get(cv2.CAP\_PROP\_FPS)

total\_frames = int(cap.get(cv2.CAP\_PROP\_FRAME\_COUNT))

print(f"视频信息: FPS={fps}, 总帧数={total\_frames}")

# 创建日志文件

log\_file = video\_path.replace('.mp4', '.log').replace('.mov', '.log')

frame\_count = 0

success\_count = 0

with open(log\_file, 'w', encoding='utf-8') as f:

f.write("frame\_number,timestamp,confidence\n")

while True:

ret, frame = cap.read()

if not ret:

break

# 每10帧处理一次（提高效率）

if frame\_count % 10 == 0:

# 转换为灰度图

gray = cv2.cvtColor(frame, cv2.COLOR\_BGR2GRAY)

# 识别数字区域（根据实际时间戳位置调整）

# 这里需要根据实际测试视频调整ROI坐标

height, width = gray.shape

roi = gray[50:100, width-200:width-50] # 右上角区域

# 使用OCR识别

try:

timestamp = pytesseract.image\_to\_string(roi, config='--psm 7 -c tessedit\_char\_whitelist=0123456789:.')

timestamp = timestamp.strip()

if timestamp and len(timestamp) > 0:

f.write(f"{frame\_count},{timestamp},1.0\n")

success\_count += 1

print(f"帧 {frame\_count}: 识别到时间戳 {timestamp}")

except Exception as e:

print(f"帧 {frame\_count}: OCR识别失败 - {e}")

frame\_count += 1

# 进度显示

if frame\_count % 100 == 0:

progress = (frame\_count / total\_frames) \* 100

print(f"处理进度: {progress:.1f}%")

cap.release()

print(f"处理完成！成功识别 {success\_count} 个时间戳")

print(f"日志文件: {log\_file}")

python

import pandas as pd

import matplotlib.pyplot as plt

import numpy as np

import sys

import os

def analyze\_delay(log\_file):

"""

分析延迟日志文件

"""

print(f"开始分析日志文件: {log\_file}")

# 读取日志文件

try:

df = pd.read\_csv(log\_file, encoding='utf-8')

except UnicodeDecodeError:

try:

df = pd.read\_csv(log\_file, encoding='gbk')

except:

print("错误：无法读取日志文件，请检查编码格式")

return None

if df.empty:

print("错误：日志文件为空")

return None

# 转换时间戳为数值（毫秒）

def parse\_timestamp(ts):

try:

# 处理各种时间格式

if ':' in str(ts):

parts = str(ts).split(':')

if len(parts) == 3: # HH:MM:SS

return int(parts[0])\*3600000 + int(parts[1])\*60000 + float(parts[2])\*1000

elif len(parts) == 2: # MM:SS

return int(parts[0])\*60000 + float(parts[1])\*1000

else:

return float(ts) \* 1000 # 假设是秒数

except:

return None

df['timestamp\_ms'] = df['timestamp'].apply(parse\_timestamp)

df = df.dropna(subset=['timestamp\_ms'])

if df.empty:

print("错误：没有有效的时间戳数据")

return None

# 计算延迟（假设第一个时间戳为基准）

base\_time = df['timestamp\_ms'].iloc[0]

df['delay\_ms'] = df['timestamp\_ms'] - base\_time

# 计算帧时间（基于帧率估算）

# 假设标准帧率，实际应根据视频调整

assumed\_fps = 30

df['expected\_time'] = df['frame\_number'] \* (1000 / assumed\_fps)

df['actual\_delay'] = df['delay\_ms'] - df['expected\_time']

# 计算统计信息

delays = df['actual\_delay'].dropna()

stats = {

'数据点数': len(delays),

'平均值': np.mean(delays),

'中位数': np.median(delays),

'最小值': np.min(delays),

'最大值': np.max(delays),

'标准差': np.std(delays),

'95分位数': np.percentile(delays, 95)

}

# 输出统计结果

print("\n=== 延迟统计结果 ===")

for key, value in stats.items():

if isinstance(value, float):

print(f"{key}: {value:.2f}ms")

else:

print(f"{key}: {value}")

# 生成图表

plt.figure(figsize=(12, 8))

# 延迟趋势图

plt.subplot(2, 1, 1)

plt.plot(df['frame\_number'], df['actual\_delay'], 'b-', alpha=0.7, linewidth=1)

plt.xlabel('帧号')

plt.ylabel('延迟 (ms)')

plt.title(f'视频延迟分析 - {os.path.basename(log\_file)}')

plt.grid(True, alpha=0.3)

# 延迟分布直方图

plt.subplot(2, 1, 2)

plt.hist(delays, bins=50, alpha=0.7, edgecolor='black')

plt.xlabel('延迟 (ms)')

plt.ylabel('频次')

plt.title('延迟分布直方图')

plt.grid(True, alpha=0.3)

plt.tight\_layout()

# 保存图表

chart\_file = log\_file.replace('.log', '\_chart.png')

plt.savefig(chart\_file, dpi=300, bbox\_inches='tight')

print(f"\n图表已保存: {chart\_file}")

# 保存详细结果

result\_file = log\_file.replace('.log', '\_results.txt')

with open(result\_file, 'w', encoding='utf-8') as f:

f.write("视频延迟测试结果\n")

f.write("================\n\n")

for key, value in stats.items():

if isinstance(value, float):

f.write(f"{key}: {value:.2f}ms\n")

else:

f.write(f"{key}: {value}\n")

print(f"详细结果已保存: {result\_file}")

return stats

if \_\_name\_\_ == "\_\_main\_\_":

if len(sys.argv) != 2:

print("用法: python analysis.py <日志文件路径>")

sys.exit(1)

log\_file = sys.argv[1]

if not os.path.exists(log\_file):

print(f"错误：文件 {log\_file} 不存在")

sys.exit(1)

analyze\_delay(log\_file)

def generate\_timestamp\_video():

"""生成高精度时间戳测试视频"""

output\_dir = "C:\\直播测试素材"

os.makedirs(output\_dir, exist\_ok=True)

# 视频参数

width, height = 1280, 720

fps = 30

duration = 300 # 5分钟

total\_frames = fps \* duration

# 创建视频写入对象

fourcc = cv2.VideoWriter\_fourcc(\*'mp4v')

video\_path = os.path.join(output\_dir, 'high\_precision\_timestamp.mp4')

out = cv2.VideoWriter(video\_path, fourcc, fps, (width, height))

print("生成高精度时间戳测试视频...")

for frame\_num in range(total\_frames):

# 创建黑色背景

frame = np.zeros((height, width, 3), dtype=np.uint8)

# 计算当前时间（毫秒）

current\_time\_ms = (frame\_num / fps) \* 1000

# 绘制时间戳信息

texts = [

f"TIMESTAMP: {current\_time\_ms:.1f}ms",

f"FRAME: {frame\_num}",

f"TIME: {frame\_num//fps//60:02d}:{(frame\_num//fps)%60:02d}.{(frame\_num%fps)\*33:03d}",

f"TEST PATTERN - VIDEO DELAY MEASUREMENT"

]

positions = [(50, 100), (50, 160), (50, 220), (50, 50)]

colors = [(255, 255, 255), (255, 200, 100), (200, 255, 200), (100, 200, 255)]

sizes = [1.5, 1.0, 0.8, 1.2]

for i, text in enumerate(texts):

cv2.putText(frame, text, positions[i], cv2.FONT\_HERSHEY\_SIMPLEX,

sizes[i], colors[i], 2)

# 添加移动元素用于视觉确认

moving\_x = (frame\_num \* 2) % (width - 200) + 100

cv2.circle(frame, (moving\_x, 500), 30, (0, 255, 0), -1)

cv2.putText(frame, "MOVING MARKER", (moving\_x-80, 550),

cv2.FONT\_HERSHEY\_SIMPLEX, 0.7, (0, 255, 0), 2)

# 添加数字时钟（每100ms更新）

if frame\_num % 3 == 0: # 每3帧更新一次（约100ms）

clock\_time = f"CLOCK: {(current\_time\_ms//100)%10:01d}"

cv2.putText(frame, clock\_time, (width-200, 100),

cv2.FONT\_HERSHEY\_SIMPLEX, 1, (255, 100, 100), 2)

out.write(frame)

if frame\_num % 300 == 0: # 每10秒显示进度

progress = (frame\_num / total\_frames) \* 100

print(f"生成进度: {progress:.1f}%")

out.release()

print(f"视频生成完成: {video\_path}")

return video\_path

"White bar should move smoothly"

]

for i, text in enumerate(info\_texts):

cv2.putText(frame, text, (50, 100 + i\*40),

cv2.FONT\_HERSHEY\_SIMPLEX, 0.8, (0, 0, 0), 2)

# 添加网格用于视觉检测卡顿

for x in range(0, width, 50):
